local function readCachedRoute()
    if not (isfile and readfile and isfile(KEY_CACHE_FILE)) then return nil end
    local ok, data = pcall(function()
        return game:GetService("HttpService"):JSONDecode(readfile(KEY_CACHE_FILE))
    end)
    if not ok or type(data) ~= "table" then return nil end
    if type(data.route) ~= "number" or type(data.expiresAt) ~= "number" then return nil end
    if os.time() >= data.expiresAt then return nil end
    if data.route < 1 or data.route > #KEY_LIST then return nil end
    return data.route
end

    if not writefile then return end
    pcall(function()
        local payload = game:GetService("HttpService"):JSONEncode({
            route = route, expiresAt = os.time() + KEY_CACHE_TTL
        })
        writefile(KEY_CACHE_FILE, payload)
    end)
end

local AnimController = {
    Motors = {}, BaseC0 = {},
    State = "Idle", StateEndsAt = 0,

local function captureMotor(...)
    local names = {...}
    for _, desc in ipairs(character:GetDescendants()) do
        if desc:IsA("Motor6D") then
            for _, n in ipairs(names) do
                if desc.Name == n then return desc end
            end
        end
    end
    return nil
end

function AnimController:RefreshMotors()
    self.Motors = {
        root          = captureMotor("RootJoint","Root"),
        waist         = captureMotor("Waist"),
        neck          = captureMotor("Neck"),
        rightShoulder = captureMotor("RightShoulder","Right Shoulder"),
        leftShoulder  = captureMotor("LeftShoulder","Left Shoulder"),
        rightHip      = captureMotor("RightHip","Right Hip"),
        leftHip       = captureMotor("LeftHip","Left Hip"),
    }
    self.BaseC0 = {}
    for k, m in pairs(self.Motors) do
        if m then
            self.BaseC0[k] = m.C0
            m.Transform = CFrame.new()
        end
    end
end

function AnimController:SetState(newState, duration)
    self.State = newState
    self.StateEndsAt = duration and (os.clock() + duration) or 0
end

function AnimController:AutoState()
    if not humanoid or not root then return "Idle" end
    if airwalkActive then return "Airwalk" end
    local st = humanoid:GetState()
    if st == Enum.HumanoidStateType.Jumping  then return "Jump" end
    if st == Enum.HumanoidStateType.Freefall then
        return root.AssemblyLinearVelocity.Y > 0 and "Jump" or "Fall"
    end
    if humanoid.MoveDirection.Magnitude > 0.08 then return "Move" end
    return "Idle"
end

function AnimController:ApplyPose(targets, alpha)
    for k, motor in pairs(self.Motors) do
        if motor and self.BaseC0[k] then
            local offset = targets[k] or CFrame.new()
            motor.C0 = Clerp(motor.C0, self.BaseC0[k] * offset, alpha)
        end
    end
end

function AnimController:Update(dt)
    if not character or not root or not humanoid or humanoid.Health <= 0 then return end
    self.Sine += dt * 8
    local locked = self.StateEndsAt > os.clock()
    if not locked then self.State = self:AutoState() end

    -- airwalk physics
    if self.State == "Airwalk" then
        local cam   = workspace.CurrentCamera
        local look  = cam and cam.CFrame.LookVector or root.CFrame.LookVector
        local right = cam and cam.CFrame.RightVector or root.CFrame.RightVector
        local mi    = humanoid and humanoid.MoveDirection or Vector3.zero
        local lx    = mi:Dot(right)
        local lz    = mi:Dot(look)
        local move  = (look*(airwalkDirection.Z+lz))+(right*(airwalkDirection.X+lx))+Vector3.new(0,airwalkDirection.Y,0)
        root.AssemblyLinearVelocity = move*34 + Vector3.new(0, math.sin(os.clock()*3)*1.2, 0)
        local fl = Vector3.new(look.X,0,look.Z)
        if fl.Magnitude > 0.05 then
            root.CFrame = root.CFrame:Lerp(CFrame.new(root.Position, root.Position+fl), 0.18)
        end
    elseif self.State == "Kurohitsugi" or self.State == "Reiatsu" or self.State == "Hover" then
        root.AssemblyLinearVelocity = Vector3.new(0, math.sin(os.clock()*3)*1.5, 0)
    end

    if not locked and self.State == "Move" then return end
    local alpha = locked and 0.22 or 0.08
    local targets = self:TargetPose(self.State)
    if targets then self:ApplyPose(targets, alpha) end
end

local function installProceduralAnimationEngine()

        AnimController.Connected = true
        RunService.RenderStepped:Connect(function(dt)
            AnimController:Update(dt)
        end)
    end
end

local function cameraPulse(power, duration)
    local cam = workspace.CurrentCamera
    if not cam then return end
    local start = os.clock()
    local conn
    conn = RunService.RenderStepped:Connect(function()
        local alpha = (os.clock()-start)/duration
        if alpha >= 1 then conn:Disconnect() return end
        local f = 1-alpha
        cam.CFrame = cam.CFrame * CF(
            rng:NextNumber(-power,power)*f,
            rng:NextNumber(-power,power)*f,
            rng:NextNumber(-power,power)*f
        )
    end)
end

local function applyPostProcessing(bloom, blur, saturation, duration)
    local oldBloom = ppSettings.Bloom.Intensity
    local oldBlur  = ppSettings.Blur.Size
    local oldSat   = ppSettings.Color.Saturation
    
    tween(ppSettings.Bloom, 0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out, {Intensity = bloom})
    tween(ppSettings.Blur, 0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out, {Size = blur})
    tween(ppSettings.Color, 0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out, {Saturation = saturation})
    
    task.delay(duration, function()
        tween(ppSettings.Bloom, 0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.In, {Intensity = oldBloom})
        tween(ppSettings.Blur, 0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.In, {Size = oldBlur})
        tween(ppSettings.Color, 0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.In, {Saturation = oldSat})
    end)
end

local function beam(startPos, endPos, color, width, lifetime)

        {Size=Vector3.new(width*0.1,width*0.1,dist), Transparency=1})

end

local function sigil(cf, radius, color, lifetime, spokes)
    ring(cf, radius, color, lifetime, 0.07)
    for i = 1, spokes or 12 do
        local a  = (math.pi*2/(spokes or 12))*i
        local s0 = (cf*CF(math.cos(a)*radius*0.12,0,math.sin(a)*radius*0.12)).Position
        local s1 = (cf*CF(math.cos(a)*radius,     0,math.sin(a)*radius     )).Position
        task.delay(i*0.01, function()
            beam(s0, s1, color, 0.055, lifetime*0.72)
        end)
    end
    -- inner diamond spokes
    for i = 1, 4 do
        local a  = (math.pi*2/4)*i + math.pi/4
        local s0 = (cf*CF(math.cos(a)*radius*0.5,0,math.sin(a)*radius*0.5)).Position
        local s1 = (cf*CF(math.cos(a+math.pi/4)*radius*0.8,0,math.sin(a+math.pi/4)*radius*0.8)).Position
        task.delay(i*0.018, function()
            beam(s0, s1, GOLD, 0.04, lifetime*0.55)
        end)
    end
end

local function speedLines(origin, dir, count, lifetime)
    local right = dir:Cross(Vector3.new(0,1,0))
    if right.Magnitude < 0.1 then right = Vector3.new(1,0,0) end
    right = right.Unit
    for i = 1, count do

        local s0    = origin+off-dir.Unit*rng:NextNumber(1,4)

        task.delay(i*0.006, function()
            beam(s0, s1, Color3.new(0,0,0), rng:NextNumber(0.04,0.1), lifetime or 0.2)
        end)
    end
end

local function markAndFlingTarget(model, originPos, power)
    if not model or model == character or (model:IsA("Model") and model == Players.LocalPlayer.Character) then return end -- Ensure local player is never flung
    local eHum  = model:FindFirstChildOfClass("Humanoid")
    local eRoot = model:FindFirstChild("HumanoidRootPart")
                or model:FindFirstChild("Torso")
                or model:FindFirstChild("UpperTorso")
    if not eHum or not eRoot or eHum.Health <= 0 then return end

    -- Red flash + white border highlight
    local hl = Instance.new("Highlight")
    hl.Name              = "AizenHitMark"
    hl.FillColor         = RED_HIT
    hl.OutlineColor      = WHITE
    hl.FillTransparency  = 0.18
    hl.OutlineTransparency = 0
    hl.Parent            = model
    Debris:AddItem(hl, 0.7)

    -- Screen red flash on all visible parts
    for _, d in ipairs(model:GetDescendants()) do
        if d:IsA("BasePart") then
            local orig = d.Color
            d.Color = RED_HIT
            task.delay(0.12, function()
                if d and d.Parent then
                    tween(d, 0.22, Enum.EasingStyle.Sine, Enum.EasingDirection.Out, {Color=orig})
                end
            end)
        end
    end

    -- Fling
    local dir = (eRoot.Position - originPos)
    if dir.Magnitude < 0.1 then dir = root.CFrame.LookVector end
    print("Flinging target: " .. model.Name) -- Debugging line
    eRoot.AssemblyLinearVelocity = dir.Unit*(power or 82) + Vector3.new(0, 38, 0) -- Added a vertical component for more pronounced fling
    eHum:TakeDamage(math.max(6, eHum.MaxHealth*0.1))

    orb(eRoot.Position+Vector3.new(0,1.5,0), 1.4, RED_HIT, 0.35)
    ring(CF(eRoot.Position)*ANGLES(RAD(90),0,0), 4, WHITE, 0.28, 0.06)
    cameraPulse(0.14, 0.28)
end

-- ============================================================
local function findFrontTarget(range, arc)
    local bestModel, bestRoot, bestDist = nil, nil, range or 45
    local forward = root.CFrame.LookVector
    for _, model in ipairs(workspace:GetChildren()) do
        if model:IsA("Model") and model ~= character then
            local eHum  = model:FindFirstChildOfClass("Humanoid")
            local eRoot = model:FindFirstChild("HumanoidRootPart")
                        or model:FindFirstChild("Torso")
                        or model:FindFirstChild("UpperTorso")
            if eHum and eRoot and eHum.Health > 0 then
                local delta = eRoot.Position - root.Position
                local dist  = delta.Magnitude
                if dist <= bestDist and dist > 1 then
                    if forward:Dot(delta.Unit) >= (arc or 0.2) then
                        bestModel, bestRoot, bestDist = model, eRoot, dist
                    end
                end
            end
        end
    end
    return bestModel, bestRoot
end

local function collapseTarget(model, duration)
    local eHum  = model and model:FindFirstChildOfClass("Humanoid")
    local eRoot = model and (model:FindFirstChild("HumanoidRootPart")
                         or model:FindFirstChild("Torso")
                         or model:FindFirstChild("UpperTorso"))
    if not eHum or not eRoot then return end
    eHum:TakeDamage(math.max(4, eHum.MaxHealth*0.07))
    eHum.Sit       = true
    eHum.WalkSpeed = math.max(0, eHum.WalkSpeed*0.3)
    eRoot.AssemblyLinearVelocity = Vector3.new(0,-18,0)
    orb(eRoot.Position+Vector3.new(0,1,0), 1.4, DARK, 0.38)
    task.delay(duration or 1.2, function()
        if eHum then eHum.Sit = false end
    end)
end

local function flingFrontTarget(power)
    local model = select(1, findFrontTarget(60, 0.1))
    if model then markAndFlingTarget(model, root.Position, power or 120) end
end

local function cooldown(tool, seconds)
    if activeCooldowns[tool] then return false end
    activeCooldowns[tool] = true
    local orig = tool.Name
    task.spawn(function()
        for r = seconds, 1, -1 do
            tool.Name = string.format("%s [%ds]", orig, r)
            task.wait(1)
        end
        tool.Name = orig
        activeCooldowns[tool] = nil
    end)
    return true
end

            local behindCF = targetRoot.CFrame * CF(0,0,5.5)

                root.CFrame = root.CFrame:Lerp(CF(behindCF.Position, targetPos), i/5)
                RunService.RenderStepped:Wait()

                cameraPulse(0.1, 0.2)

                for i = 1, 14 do

                                 + Vector3.new(0,rng:NextNumber(-1,1),0)
                    task.delay(i*0.012, function()
                        beam(slashOrigin,
                             targetPos+spread+Vector3.new(0,1,0),
                             i%2==0 and GOLD or WHITE, 0.09, 0.22)
                    end)
                end

                -- X-shaped slash mark at target
                local function slashMark(pos)
                    local s = 3
                    for _, ang in ipairs({45,-45}) do
                        local r  = CF(pos)*ANGLES(0,RAD(ang),0)
                        local p0 = (r*CF(-s,1.2,0)).Position
                        local p1 = (r*CF( s,1.2,0)).Position
                        beam(p0, p1, WHITE, 0.12, 0.32)
                    end
                end
                slashMark(targetPos)

                -- Hit flash + fling
                collapseTarget(targetModel, 1.3)
                markAndFlingTarget(targetModel, root.Position, 88)

                task.delay(0.35, function()
                    -- PHASE 5: After-image bloom
                    sideMangaCaption("Behind\nyou.", 0.9)
                    petalBurst(targetPos+Vector3.new(0,2,0), 32, ACCENT, 0.85)
                    orb(targetPos+Vector3.new(0,2,0), 2.2, GOLD, 0.55)
                    for i = 1, 7 do
                        task.delay(i*0.045, function()
                            ring(CF(targetPos)*ANGLES(RAD(90),0,0), 2+i*1.8, i%2==0 and GOLD or ACCENT, 0.6, 0.07)
                        end)
                    end
                    -- Ghost linger at slash point
                    ghostClone(targetPos-root.Position, 0.85)
                    cameraPulse(0.07, 0.4)
                end)
            end)
        end)
    end)

local function hadoGroundLoop(center, duration)
    local start = os.clock()
    local conn
    conn = RunService.RenderStepped:Connect(function()
        if os.clock()-start > duration then conn:Disconnect() return end
        if rng:NextNumber() < 0.45 then
            local off  = Vector3.new(rng:NextNumber(-8,8),0,rng:NextNumber(-8,8))
            local base = center.Position+off
            beam(base, base+Vector3.new(rng:NextNumber(-1,1),rng:NextNumber(5,13),rng:NextNumber(-1,1)),

end

        tween(core, 0.28, Enum.EasingStyle.Quart, Enum.EasingDirection.In, {Transparency=1})

local function kurohitsugiInnerPrison(center, lifetime)
    local sz = 5.5
    for x = -1, 1, 2 do
        for z = -1, 1, 2 do
            beam((center*CF(x*sz/2, 3,   z*sz/2)).Position,
                 (center*CF(x*sz/2, 14.5,z*sz/2)).Position,
                 GOLD, 0.085, lifetime or 0.9)
        end
    end
    for y = 4, 14, 5 do
        local corners = {
            (center*CF(-sz/2,y,-sz/2)).Position,
            (center*CF( sz/2,y,-sz/2)).Position,
            (center*CF( sz/2,y, sz/2)).Position,
            (center*CF(-sz/2,y, sz/2)).Position,
        }
        for i = 1, #corners do
            beam(corners[i], corners[(i%#corners)+1],
                i==1 and WHITE or GOLD, 0.075, lifetime or 0.9)
        end
    end
    local captive = makePart("Captive", Vector3.new(sz,sz,sz),
        center*CF(0,8.8,0), Color3.fromRGB(9,0,16), Enum.Material.Neon, 0.1)
    tween(captive, lifetime or 0.9, Enum.EasingStyle.Sine, Enum.EasingDirection.Out,
        {Transparency=1, Size=Vector3.new(sz*0.55,sz*0.55,sz*0.55)})
    Debris:AddItem(captive, (lifetime or 0.9)+0.1)
end

-- ============================================================
local function shunpo()
    cinematicCaption("SHUNPO", "A step outside perception.", 0.6)
    sideMangaCaption("Too slow.", 0.85)
    playBodyAnimation("shunpo", 0.72)
    ghostClone(Vector3.zero, 0.58)
    cameraPulse(0.09, 0.28)
    distortionBarrier(0.3, 8, 2) -- Distortion at start of shunpo

    local startCF = root.CFrame
    local targetModel, targetRoot = findFrontTarget(44, 0.18)
    local destCF = targetRoot
        and CF((targetRoot.CFrame*CF(0,0,4.8)).Position, targetRoot.Position)
        or  (startCF + startCF.LookVector*40)

    -- Speed lines before
    speedLines(startCF.Position+Vector3.new(0,2,0), startCF.LookVector, 30, 0.25) -- Enhanced speed lines at start
    orb(startCF.Position+Vector3.new(0,2,0), 1.0, WHITE, 0.2) -- Flash at start
    -- Afterimage rings along path
    for i = 1, 12 do
        local cf = startCF:Lerp(destCF, i/12)
        task.delay(i*0.014, function()
            ring(cf*ANGLES(RAD(90),0,0), 2.5+i*0.3, ACCENT, 0.32, 0.06)
            petalBurst(cf.Position+Vector3.new(0,2,0), 3, GOLD, 0.3)
        end)
    end

    humanoid:ChangeState(Enum.HumanoidStateType.Physics)
    root.CFrame = destCF
    speedLines(root.Position+Vector3.new(0,2,0), root.CFrame.LookVector, 30, 0.25) -- Enhanced speed lines at end
    distortionBarrier(0.3, 8, 2) -- Distortion at end of shunpo
    orb(root.Position+Vector3.new(0,2,0), 1.0, WHITE, 0.2) -- Flash at end
    glassShatter(root.Position+Vector3.new(0,2,0), -root.CFrame.LookVector, 18)

    if targetModel then
        collapseTarget(targetModel, 1.05)
        markAndFlingTarget(targetModel, root.Position, 50)
    end
    task.delay(0.1, function()
        if humanoid then humanoid:ChangeState(Enum.HumanoidStateType.Running) end
    end)
end

-- ============================================================
local function reiatsuCrush()
    cutsceneBars(0.9)
    sideMangaCaption("Do not\napproach.", 1.15)
    playBodyAnimation("reiatsu", 0.95)
    local rightHand = character:FindFirstChild("RightHand") or character:FindFirstChild("Right Arm")
    local leftHand  = character:FindFirstChild("LeftHand")  or character:FindFirstChild("Left Arm")
    if rightHand then handGlowParticles(rightHand, GOLD, 0.95) end
    if leftHand  then handGlowParticles(leftHand, ACCENT, 0.95) end
    distortionBarrier(1.0, 17, 7)
    cameraPulse(0.17, 0.58)

    for i = 1, 12 do
        task.delay(i*0.052, function()
            ring(root.CFrame*ANGLES(RAD(90),0,0), 3+i*2.6, i%2==0 and GOLD or ACCENT, 0.65, 0.15)
        end)
    end
    local aura = orb(root.Position+Vector3.new(0,1.8,0), 4.8, ACCENT, 1.0)
    aura.Transparency = 0.52
    petalBurst(root.Position+Vector3.new(0,2,0), 46, GOLD, 1.0)

    -- Crush all nearby
    for _, model in ipairs(workspace:GetChildren()) do
        if model:IsA("Model") and model ~= character then
            local eHum  = model:FindFirstChildOfClass("Humanoid")
            local eRoot = model:FindFirstChild("HumanoidRootPart")
                       or model:FindFirstChild("Torso")
                       or model:FindFirstChild("UpperTorso")
            if eHum and eRoot and eHum.Health > 0 then
                local dist = (eRoot.Position-root.Position).Magnitude
                if dist <= 20 then
                    collapseTarget(model, 1.35)
                    markAndFlingTarget(model, root.Position, 70)
                end
            end
        end
    end
end

local airwalkGui = createAirwalkControls()

local function setAirwalk(enabled)
    airwalkActive   = enabled
    airwalkDirection= Vector3.zero
    airwalkGui.Enabled = enabled
    if enabled then
        AnimController:SetState("Airwalk")
        humanoid.PlatformStand = true
        sideMangaCaption("Airwalk\nenabled", 1.0)
        distortionBarrier(0.45, 10, 3)
    else
        humanoid.PlatformStand = false
        airwalkDirection       = Vector3.zero
        airwalkVerticalMode    = 0
        if root then root.AssemblyLinearVelocity = Vector3.zero end
        AnimController:SetState("Idle")
        sideMangaCaption("Airwalk\noff", 0.6)
    end
end

local function installAirwalkControls()
    UserInputService.JumpRequest:Connect(function()
        local now = os.clock()
        if now - lastJumpRequest < 0.35 then
            if not airwalkActive then
                setAirwalk(true)
            else
                airwalkVerticalMode = (airwalkVerticalMode+1)%3
                airwalkDirection = Vector3.new(
                    airwalkDirection.X,
                    airwalkVerticalMode==1 and 1 or (airwalkVerticalMode==2 and -1 or 0),
                    airwalkDirection.Z
                )
            end
        end
        lastJumpRequest = now
    end)

    UserInputService.InputBegan:Connect(function(inp, proc)
        if proc or not airwalkActive then return end
        if inp.KeyCode == Enum.KeyCode.W then airwalkDirection += Vector3.new(0,0,1)
        elseif inp.KeyCode == Enum.KeyCode.S then airwalkDirection += Vector3.new(0,0,-1)
        elseif inp.KeyCode == Enum.KeyCode.E then airwalkDirection += Vector3.new(0,1,0)
        elseif inp.KeyCode == Enum.KeyCode.Q then airwalkDirection += Vector3.new(0,-1,0)
        end
    end)
    UserInputService.InputEnded:Connect(function(inp)
        if not airwalkActive then return end
        if inp.KeyCode == Enum.KeyCode.W then airwalkDirection -= Vector3.new(0,0,1)
        elseif inp.KeyCode == Enum.KeyCode.S then airwalkDirection -= Vector3.new(0,0,-1)
        elseif inp.KeyCode == Enum.KeyCode.E then airwalkDirection -= Vector3.new(0,1,0)
        elseif inp.KeyCode == Enum.KeyCode.Q then airwalkDirection -= Vector3.new(0,-1,0)
        end
    end)
end

-- KEYBOARD
-- ============================================================
local function installKeyboard()
    UserInputService.InputBegan:Connect(function(inp, proc)
        if proc or not root then return end
        if inp.KeyCode == Enum.KeyCode.Z then kyokaSuigetsu()
        elseif inp.KeyCode == Enum.KeyCode.X then kurohitsugi()
        elseif inp.KeyCode == Enum.KeyCode.C then raikoho()
        elseif inp.KeyCode == Enum.KeyCode.V then shunpo()
        elseif inp.KeyCode == Enum.KeyCode.B then reiatsuCrush()
        elseif inp.KeyCode == Enum.KeyCode.LeftShift then setShiftLock(not shiftLockEnabled)
        elseif inp.KeyCode == Enum.KeyCode.F and airwalkActive then setAirwalk(false)
        end
    end)
end

-- PASSIVE DOMINION
-- ============================================================
local function installPassiveDominion()
    local damaged = {}
    local passiveTimer = 0
    RunService.Heartbeat:Connect(function()
        if not root or not root.Parent or not humanoid or humanoid.Health <= 0 then return end
        local threat = false
        for _, model in ipairs(workspace:GetChildren()) do
            if model:IsA("Model") and model ~= character then
                local eHum  = model:FindFirstChildOfClass("Humanoid")
                local eRoot = model:FindFirstChild("HumanoidRootPart")
                           or model:FindFirstChild("Torso")
                           or model:FindFirstChild("UpperTorso")
                if eHum and eRoot and eHum.Health > 0 then
                    local off    = eRoot.Position - root.Position
                    local dist   = off.Magnitude
                    local front  = dist>0.1 and root.CFrame.LookVector:Dot(off.Unit)>0.2
                    if dist<=24 or (front and dist<=35) then
                        threat = true
                        if os.clock()-(damaged[eHum] or 0) > 0.68 then
                            damaged[eHum] = os.clock()
                            collapseTarget(model, 1.3)
                            markAndFlingTarget(model, root.Position, front and 40 or 68)
                            beam(root.Position+Vector3.new(0,2,0),
                                 eRoot.Position+Vector3.new(0,1.5,0), ACCENT, 0.15, 0.22)
                            sideMangaCaption("Kneel.", 0.5)
                        end
                    end
                end
            end
        end
        if threat and os.clock()-passiveTimer > 1.3 then
            passiveTimer = os.clock()
            distortionBarrier(0.55, 14, 3)
        end
    end)
end

local function createTool(name, cooldownSeconds, callback)
    local tool = Instance.new("Tool")
    tool.Name         = name
    tool.ToolTip      = KIT_NAME.." | "..name
    tool.TextureId    = ""
    tool.RequiresHandle = false
    tool.CanBeDropped = false
    tool.Parent       = backpack
    tool.Activated:Connect(function()
        if not character or not root or humanoid.Health <= 0 then return end
        if cooldown(tool, cooldownSeconds) then callback(tool) end
    end)
    return tool
end
